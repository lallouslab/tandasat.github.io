155 _Use_decl_annotations_ bool EptIsEptAvailable() {
156  PAGED_CODE();
157 
158  int regs[4] = {};
159  __cpuidex(regs, 0x80000008, 0);
160  Cpuid80000008Eax cpuidEax = {static_cast<ULONG32>(regs[0])};
161  HYPERPLATFORM_LOG_DEBUG("Physical Address Range = %d bits",
162  cpuidEax.fields.physical_address_bits);
163 
164  // No processors supporting the Intel 64 architecture support more than 48
165  // physical-address bits
166  if (cpuidEax.fields.physical_address_bits > 48) {
167  return false;
168  }
169 
170  // page walk length is 4 steps
171  // extended page tables can be laid out in write-back memory
172  // INVEPT instruction with all possible types is supported
173  Ia32VmxEptVpidCapMsr vpid = {UtilReadMsr64(Msr::kIa32VmxEptVpidCap)};
174  if (!vpid.fields.support_page_walk_length4 ||
175  !vpid.fields.support_write_back_memory_type ||
176  !vpid.fields.support_invept ||
177  !vpid.fields.support_single_context_invept ||
178  !vpid.fields.support_all_context_invept) {
179  return false;
180  }
181  return true;
182 }

188 _Use_decl_annotations_ EptData *EptInitialization() {
189  PAGED_CODE();
190 
191  static const auto kEptPageWalkLevel = 4ul;
192 
193  // Allocate ept_data
194  const auto ept_data = reinterpret_cast<EptData *>(ExAllocatePoolWithTag(
195  NonPagedPoolNx, sizeof(EptData), kHyperPlatformCommonPoolTag));
196  if (!ept_data) {
197  return nullptr;
198  }
199  RtlZeroMemory(ept_data, sizeof(EptData));
200 
201  // Allocate EptPointer
202  const auto ept_poiner = reinterpret_cast<EptPointer *>(ExAllocatePoolWithTag(
203  NonPagedPoolNx, PAGE_SIZE, kHyperPlatformCommonPoolTag));
204  if (!ept_poiner) {
205  ExFreePoolWithTag(ept_data, kHyperPlatformCommonPoolTag);
206  return nullptr;
207  }
208  RtlZeroMemory(ept_poiner, PAGE_SIZE);
209 
210  // Allocate EPT_PML4 and initialize EptPointer
211  const auto ept_pml4 =
212  reinterpret_cast<EptCommonEntry *>(ExAllocatePoolWithTag(
213  NonPagedPoolNx, PAGE_SIZE, kHyperPlatformCommonPoolTag));
214  if (!ept_pml4) {
215  ExFreePoolWithTag(ept_poiner, kHyperPlatformCommonPoolTag);
216  ExFreePoolWithTag(ept_data, kHyperPlatformCommonPoolTag);
217  return nullptr;
218  }
219  RtlZeroMemory(ept_pml4, PAGE_SIZE);
220  ept_poiner->fields.memory_type =
221  static_cast<ULONG64>(memory_type::kWriteBack);
222  ept_poiner->fields.page_walk_length = kEptPageWalkLevel - 1;
223  ept_poiner->fields.pml4_address = UtilPfnFromPa(UtilPaFromVa(ept_pml4));
224 
225  // Initialize all EPT entries for all physical memory pages
226  const auto pm_ranges = UtilGetPhysicalMemoryRanges();
227  for (auto run_index = 0ul; run_index < pm_ranges->number_of_runs;
228  ++run_index) {
229  const auto run = &pm_ranges->run[run_index];
230  const auto base_addr = run->base_page * PAGE_SIZE;
231  for (auto page_index = 0ull; page_index < run->page_count; ++page_index) {
232  const auto indexed_addr = base_addr + page_index * PAGE_SIZE;
233  const auto ept_pt_entry =
234  EptpConstructTables(ept_pml4, 4, indexed_addr, nullptr);
235  if (!ept_pt_entry) {
236  EptpDestructTables(ept_pml4, 4);
237  ExFreePoolWithTag(ept_poiner, kHyperPlatformCommonPoolTag);
238  ExFreePoolWithTag(ept_data, kHyperPlatformCommonPoolTag);
239  return nullptr;
240  }
241  }
242  }
243 
244  // Initialize an EPT entry for APIC_BASE. It is required to allocated it now
245  // for some reasons, or else, system hangs.
246  const Ia32ApicBaseMsr apic_msr = {UtilReadMsr64(Msr::kIa32ApicBase)};
247  if (!EptpConstructTables(ept_pml4, 4, apic_msr.fields.apic_base * PAGE_SIZE,
248  nullptr)) {
249  EptpDestructTables(ept_pml4, 4);
250  ExFreePoolWithTag(ept_poiner, kHyperPlatformCommonPoolTag);
251  ExFreePoolWithTag(ept_data, kHyperPlatformCommonPoolTag);
252  return nullptr;
253  }
254 
255  // Allocate preallocated_entries
256  const auto preallocated_entries_size =
257  sizeof(EptCommonEntry *) * kVmxpNumberOfPreallocatedEntries;
258  const auto preallocated_entries = reinterpret_cast<EptCommonEntry **>(
259  ExAllocatePoolWithTag(NonPagedPoolNx, preallocated_entries_size,
260  kHyperPlatformCommonPoolTag));
261  if (!preallocated_entries) {
262  EptpDestructTables(ept_pml4, 4);
263  ExFreePoolWithTag(ept_poiner, kHyperPlatformCommonPoolTag);
264  ExFreePoolWithTag(ept_data, kHyperPlatformCommonPoolTag);
265  return nullptr;
266  }
267  RtlZeroMemory(preallocated_entries, preallocated_entries_size);
268 
269  // And fill preallocated_entries with newly created entries
270  for (auto i = 0ul; i < kVmxpNumberOfPreallocatedEntries; ++i) {
271  const auto ept_entry = EptpAllocateEptEntry(nullptr);
272  if (!ept_entry) {
273  EptpFreeUnusedPreAllocatedEntries(preallocated_entries, 0);
274  EptpDestructTables(ept_pml4, 4);
275  ExFreePoolWithTag(ept_poiner, kHyperPlatformCommonPoolTag);
276  ExFreePoolWithTag(ept_data, kHyperPlatformCommonPoolTag);
277  return nullptr;
278  }
279  preallocated_entries[i] = ept_entry;
280  }
281 
282  // Initialization completed
283  ept_data->ept_pointer = ept_poiner;
284  ept_data->ept_pml4 = ept_pml4;
285  ept_data->preallocated_entries = preallocated_entries;
286  ept_data->preallocated_entries_count = 0;
287  return ept_data;
288 }

290 // Allocate and initialize all EPT entries associated with the physical_address
291 _Use_decl_annotations_ static EptCommonEntry *EptpConstructTables(
292  EptCommonEntry *table, ULONG table_level, ULONG64 physical_address,
293  EptData *ept_data) {
294  switch (table_level) {
295  case 4: {
296  // table == PML4 (512 GB)
297  const auto pxe_index = EptpAddressToPxeIndex(physical_address);
298  const auto ept_pml4_entry = &table[pxe_index];
299  if (!ept_pml4_entry->all) {
300  const auto ept_pdpt = EptpAllocateEptEntry(ept_data);
301  if (!ept_pdpt) {
302  return nullptr;
303  }
304  EptpInitTableEntry(ept_pml4_entry, table_level, UtilPaFromVa(ept_pdpt));
305  }
306  return EptpConstructTables(
307  reinterpret_cast<EptCommonEntry *>(
308  UtilVaFromPfn(ept_pml4_entry->fields.physial_address)),
309  table_level - 1, physical_address, ept_data);
310  }
311  case 3: {
312  // table == PDPT (1 GB)
313  const auto ppe_index = EptpAddressToPpeIndex(physical_address);
314  const auto ept_pdpt_entry = &table[ppe_index];
315  if (!ept_pdpt_entry->all) {
316  const auto ept_pdt = EptpAllocateEptEntry(ept_data);
317  if (!ept_pdt) {
318  return nullptr;
319  }
320  EptpInitTableEntry(ept_pdpt_entry, table_level, UtilPaFromVa(ept_pdt));
321  }
322  return EptpConstructTables(
323  reinterpret_cast<EptCommonEntry *>(
324  UtilVaFromPfn(ept_pdpt_entry->fields.physial_address)),
325  table_level - 1, physical_address, ept_data);
326  }
327  case 2: {
328  // table == PDT (2 MB)
329  const auto pde_index = EptpAddressToPdeIndex(physical_address);
330  const auto ept_pdt_entry = &table[pde_index];
331  if (!ept_pdt_entry->all) {
332  const auto ept_pt = EptpAllocateEptEntry(ept_data);
333  if (!ept_pt) {
334  return nullptr;
335  }
336  EptpInitTableEntry(ept_pdt_entry, table_level, UtilPaFromVa(ept_pt));
337  }
338  return EptpConstructTables(
339  reinterpret_cast<EptCommonEntry *>(
340  UtilVaFromPfn(ept_pdt_entry->fields.physial_address)),
341  table_level - 1, physical_address, ept_data);
342  }
343  case 1: {
344  // table == PT (4 KB)
345  const auto pte_index = EptpAddressToPteIndex(physical_address);
346  const auto ept_pt_entry = &table[pte_index];
347  NT_ASSERT(!ept_pt_entry->all);
348  EptpInitTableEntry(ept_pt_entry, table_level, physical_address);
349  return ept_pt_entry;
350  }
351  default:
352  HYPERPLATFORM_COMMON_DBG_BREAK();
353  return nullptr;
354  }
355 }

434 // Deal with EPT violation VM-exit.
435 _Use_decl_annotations_ void EptHandleEptViolation(EptData *ept_data) {
436  const EptViolationQualification exit_qualification = {
437  UtilVmRead(VmcsField::kExitQualification)};
438 
439  const auto fault_pa = UtilVmRead64(VmcsField::kGuestPhysicalAddress);
440  const auto fault_va =
441  exit_qualification.fields.valid_guest_linear_address
442  ? reinterpret_cast<void *>(UtilVmRead(VmcsField::kGuestLinearAddress))
443  : nullptr;
444 
445  if (!exit_qualification.fields.ept_readable &&
446  !exit_qualification.fields.ept_writeable &&
447  !exit_qualification.fields.ept_executable) {
448  // EPT entry miss. It should be device memory.
449  HYPERPLATFORM_PERFORMANCE_MEASURE_THIS_SCOPE();
450  // HYPERPLATFORM_LOG_DEBUG_SAFE(
451  // "[INIT] Dev VA = %p, PA = %016llx, Used = %d",
452  // 0, fault_pa, ept_data->preallocated_entries_count);
453 
454  if (!IsReleaseBuild()) {
455  const auto is_device_memory = EptpIsDeviceMemory(fault_pa);
456  NT_ASSERT(is_device_memory);
457  UNREFERENCED_PARAMETER(is_device_memory);
458  }
459 
460  // There is a race condition here. If multiple processors reach this code
461  // with the same fault_pa, this function may create multiple EPT entries for
462  // one physical address and leads memory leak. This call should probably be
463  // guarded by a spin-lock but is not yet just because impact is so small.
464  EptpConstructTables(ept_data->ept_pml4, 4, fault_pa, ept_data);
465 
466  UtilInveptAll();
467 
468  } else {
469  HYPERPLATFORM_LOG_DEBUG_SAFE("[IGNR] OTH VA = %p, PA = %016llx", fault_va,
470  fault_pa);
471  }
472 }

474 // Returns if the physical_address is device memory (which could not have a
475 // corresponding PFN entry)
476 _Use_decl_annotations_ static bool EptpIsDeviceMemory(
477  ULONG64 physical_address) {
478  const auto pm_ranges = UtilGetPhysicalMemoryRanges();
479  for (auto i = 0ul; i < pm_ranges->number_of_runs; ++i) {
480  const auto current_run = &pm_ranges->run[i];
481  const auto base_addr =
482  static_cast<ULONG64>(current_run->base_page) * PAGE_SIZE;
483  const auto endAddr = base_addr + current_run->page_count * PAGE_SIZE - 1;
484  if (UtilIsInBounds(physical_address, base_addr, endAddr)) {
485  return false;
486  }
487  }
488  return true;
489 }

560 // Frees all used EPT entries by walking through whole EPT
561 _Use_decl_annotations_ static void EptpDestructTables(EptCommonEntry *table,
562  ULONG table_level) {
563  for (auto i = 0ul; i < 512; ++i) {
564  const auto entry = table[i];
565  if (entry.fields.physial_address) {
566  const auto sub_table = reinterpret_cast<EptCommonEntry *>(
567  UtilVaFromPfn(entry.fields.physial_address));
568 
569  switch (table_level) {
570  case 4: // table == PML4, sub_table == PDPT
571  case 3: // table == PDPT, sub_table == PDT
572  EptpDestructTables(sub_table, table_level - 1);
573  break;
574  case 2: // table == PDT, sub_table == PT
575  ExFreePoolWithTag(sub_table, kHyperPlatformCommonPoolTag);
576  break;
577  default:
578  HYPERPLATFORM_COMMON_DBG_BREAK();
579  break;
580  }
581  }
582  }
583  ExFreePoolWithTag(table, kHyperPlatformCommonPoolTag);
