149 LogInitialization(ULONG flag, const wchar_t *log_file_path) {
150  PAGED_CODE();
151 
152  auto status = STATUS_SUCCESS;
153 
154  g_logp_debug_flag = flag;
155 
156  // Initialize a log file if a log file path is specified.
157  bool need_reinitialization = false;
158  if (log_file_path) {
159  status = LogpInitializeBufferInfo(log_file_path, &g_logp_log_buffer_info);
160  if (status == STATUS_REINITIALIZATION_NEEDED) {
161  need_reinitialization = true;

163  return status;
164  }
165  }
166

170  if (!NT_SUCCESS(status)) {

173  HYPERPLATFORM_LOG_DEBUG("Info= %p, Buffer= %p %p, File= %S",
174  &g_logp_log_buffer_info,

176  g_logp_log_buffer_info.log_buffer2, log_file_path);
177  return (need_reinitialization ? STATUS_REINITIALIZATION_NEEDED
178  : STATUS_SUCCESS);

187 // Initialize a log file related code such as a flushing thread.
188 _Use_decl_annotations_ static NTSTATUS LogpInitializeBufferInfo(
189  const wchar_t *log_file_path, LogBufferInfo *info) {
190  PAGED_CODE();
191  NT_ASSERT(log_file_path);
192  NT_ASSERT(info);
193 
194  KeInitializeSpinLock(&info->spin_lock);
195 
196  auto status = RtlStringCchCopyW(
197  info->log_file_path, RTL_NUMBER_OF_FIELD(LogBufferInfo, log_file_path),
198  log_file_path);
199  if (!NT_SUCCESS(status)) {
200  return status;
201  }
202 
203  status = ExInitializeResourceLite(&info->resource);
204  if (!NT_SUCCESS(status)) {
205  return status;
206  }
207  info->resource_initialized = true;
208 
209  // Allocate two log buffers on NonPagedPool.
210  info->log_buffer1 = reinterpret_cast<char *>(
211  ExAllocatePoolWithTag(NonPagedPoolNx, kLogpBufferSize, kLogpPoolTag));
212  if (!info->log_buffer1) {
213  LogpFinalizeBufferInfo(info);
214  return STATUS_INSUFFICIENT_RESOURCES;
215  }
216 
217  info->log_buffer2 = reinterpret_cast<char *>(
218  ExAllocatePoolWithTag(NonPagedPoolNx, kLogpBufferSize, kLogpPoolTag));
219  if (!info->log_buffer2) {
220  LogpFinalizeBufferInfo(info);
221  return STATUS_INSUFFICIENT_RESOURCES;
222  }
223 
224  // Initialize these buffers
225  RtlFillMemory(info->log_buffer1, kLogpBufferSize, 0xff); // for diagnostic
226  info->log_buffer1[0] = '\0';
227  info->log_buffer1[kLogpBufferSize - 1] = '\0'; // at the end
228 
229  RtlFillMemory(info->log_buffer2, kLogpBufferSize, 0xff); // for diagnostic
230  info->log_buffer2[0] = '\0';
231  info->log_buffer2[kLogpBufferSize - 1] = '\0'; // at the end
232 
233  // Buffer should be used is log_buffer1, and location should be written logs
234  // is the head of the buffer.
235  info->log_buffer_head = info->log_buffer1;
236  info->log_buffer_tail = info->log_buffer1;
237 
238  status = LogpInitializeLogFile(info);
239  if (status == STATUS_OBJECT_PATH_NOT_FOUND) {
240  HYPERPLATFORM_LOG_INFO("The log file needs to be activated later.");
241  status = STATUS_REINITIALIZATION_NEEDED;
242  } else if (!NT_SUCCESS(status)) {
243  LogpFinalizeBufferInfo(info);
244  }
245  return status;
246 }

248 // Initializes a log file and startes a log buffer thread.
249 _Use_decl_annotations_ static NTSTATUS LogpInitializeLogFile(
250  LogBufferInfo *info) {
251  PAGED_CODE();
252 
253  if (info->log_file_handle) {
254  return STATUS_SUCCESS;
255  }
256 
257  // Initialize a log file
258  UNICODE_STRING log_file_path_u = {};
259  RtlInitUnicodeString(&log_file_path_u, info->log_file_path);
260 
261  OBJECT_ATTRIBUTES oa = {};
262  InitializeObjectAttributes(&oa, &log_file_path_u,
263  OBJ_KERNEL_HANDLE | OBJ_CASE_INSENSITIVE, nullptr,
264  nullptr);
265 
266  IO_STATUS_BLOCK io_status = {};
267  auto status = ZwCreateFile(
268  &info->log_file_handle, FILE_APPEND_DATA | SYNCHRONIZE, &oa, &io_status,
269  nullptr, FILE_ATTRIBUTE_NORMAL, FILE_SHARE_READ, FILE_OPEN_IF,
270  FILE_SYNCHRONOUS_IO_NONALERT | FILE_NON_DIRECTORY_FILE, nullptr, 0);
271  if (!NT_SUCCESS(status)) {
272  return status;
273  }
274 
275  // Initialize a log buffer flush thread.
276  info->buffer_flush_thread_should_be_alive = true;
277  status = PsCreateSystemThread(&info->buffer_flush_thread_handle, GENERIC_ALL,
278  nullptr, nullptr, nullptr,
279  LogpBufferFlushThreadRoutine, info);
280  if (!NT_SUCCESS(status)) {
281  ZwClose(info->log_file_handle);
282  info->log_file_handle = nullptr;
283  info->buffer_flush_thread_should_be_alive = false;
284  return status;
285  }
286 
287  // Wait until the thead has started
288  while (!info->buffer_flush_thread_started) {
289  LogpSleep(100);
290  }
291  return status;
292 }

348  PAGED_CODE();

376

378  ExDeleteResourceLite(&info->resource);

384 _Use_decl_annotations_ NTSTATUS LogpPrint(ULONG level,
385  const char *function_name,
386  const char *format, ...) {
387  auto status = STATUS_SUCCESS;
388 
389  if (!LogpIsLogNeeded(level)) {
390  return status;
391  }
392 
393  va_list args;
394  va_start(args, format);
395  char log_message[412];
396  status = RtlStringCchVPrintfA(log_message, RTL_NUMBER_OF(log_message), format,
397  args);
398  va_end(args);
399  if (!NT_SUCCESS(status)) {
400  return status;
401  }
402  if (log_message[0] == '\0') {
403  return STATUS_INVALID_PARAMETER;
404  }
405 
406  const auto pure_level = level & 0xf0;
407  const auto attribute = level & 0x0f;
408 
409  // A single entry of log should not exceed 512 bytes. See
410  // Reading and Filtering Debugging Messages in MSDN for details.
411  char message[512];
412  static_assert(RTL_NUMBER_OF(message) <= 512,
413  "One log message should not exceed 512 bytes.");
414  status = LogpMakePrefix(pure_level, function_name, log_message, message,
415  RTL_NUMBER_OF(message));
416  if (!NT_SUCCESS(status)) {
417  return status;
418  }
419 
420  return LogpPut(message, attribute);
421 }

423 // Concatenates meta information such as the current time and a process ID to
424 // user given log message.
425 _Use_decl_annotations_ static NTSTATUS LogpMakePrefix(
426  ULONG level, const char *function_name, const char *log_message,
427  char *log_buffer, SIZE_T log_buffer_length) {
428  char const *level_string = nullptr;
429  switch (level) {
430  case kLogpLevelDebug:
431  level_string = "DBG\t";
432  break;
433  case kLogpLevelInfo:
434  level_string = "INF\t";
435  break;
436  case kLogpLevelWarn:
437  level_string = "WRN\t";
438  break;
439  case kLogpLevelError:
440  level_string = "ERR\t";
441  break;
442  default:
443  return STATUS_INVALID_PARAMETER;
444  }
445 
446  auto status = STATUS_SUCCESS;
447 
448  char time_buffer[20] = {};
449  if ((g_logp_debug_flag & kLogOptDisableTime) == 0) {
450  // Want the current time.
451  TIME_FIELDS time_fields;
452  LARGE_INTEGER system_time, local_time;
453  KeQuerySystemTime(&system_time);
454  ExSystemTimeToLocalTime(&system_time, &local_time);
455  RtlTimeToTimeFields(&local_time, &time_fields);
456 
457  status = RtlStringCchPrintfA(time_buffer, RTL_NUMBER_OF(time_buffer),
458  "%02u:%02u:%02u.%03u\t", time_fields.Hour,
459  time_fields.Minute, time_fields.Second,
460  time_fields.Milliseconds);
461  if (!NT_SUCCESS(status)) {
462  return status;
463  }
464  }
465 
466  // Want the function name
467  char function_name_buffer[50] = {};
468  if ((g_logp_debug_flag & kLogOptDisableFunctionName) == 0) {
469  const auto base_function_name = LogpFindBaseFunctionName(function_name);
470  status = RtlStringCchPrintfA(function_name_buffer,
471  RTL_NUMBER_OF(function_name_buffer), "%-40s\t",
472  base_function_name);
473  if (!NT_SUCCESS(status)) {
474  return status;
475  }
476  }
477 
478  // Want the processor number
479  char processro_number[10] = {};
480  if ((g_logp_debug_flag & kLogOptDisableProcessorNumber) == 0) {
481  status =
482  RtlStringCchPrintfA(processro_number, RTL_NUMBER_OF(processro_number),
483  "#%lu\t", KeGetCurrentProcessorNumber());
484  if (!NT_SUCCESS(status)) {
485  return status;
486  }
487  }
488 
489  //
490  // It uses PsGetProcessId(PsGetCurrentProcess()) instead of
491  // PsGetCurrentThreadProcessId() because the later sometimes returns
492  // unwanted value, for example:
493  // PID == 4 but its image name != ntoskrnl.exe
494  // The author is guessing that it is related to attaching processes but
495  // not quite sure. The former way works as expected.
496  //
497  status = RtlStringCchPrintfA(
498  log_buffer, log_buffer_length, "%s%s%s%5Iu\t%5Iu\t%-15s\t%s%s\r\n",
499  time_buffer, level_string, processro_number,
500  reinterpret_cast<ULONG_PTR>(PsGetProcessId(PsGetCurrentProcess())),
501  reinterpret_cast<ULONG_PTR>(PsGetCurrentThreadId()),
502  PsGetProcessImageFileName(PsGetCurrentProcess()), function_name_buffer,
503  log_message);
504  return status;
505 }

508 // NamespaceName::ClassName::MethodName will be returned as MethodName.

525 // Logs the entry according to attribute and the thread condition.
526 _Use_decl_annotations_ static NTSTATUS LogpPut(char *message, ULONG attribute) {
527  auto status = STATUS_SUCCESS;
528 
529  auto do_DbgPrint = ((attribute & kLogpLevelOptSafe) == 0 &&
530  KeGetCurrentIrql() < CLOCK_LEVEL);
531 
532  // Log the entry to a file or buffer.
533  auto &info = g_logp_log_buffer_info;
534  if (LogpIsLogFileEnabled(info)) {
535  // Can it log it to a file now?
536  if (((attribute & kLogpLevelOptSafe) == 0) &&
537  KeGetCurrentIrql() == PASSIVE_LEVEL && LogpIsLogFileActivated(info)) {
538 #pragma warning(push)
539 #pragma warning(disable : 28123)
540  if (!KeAreAllApcsDisabled()) {
541  // Yes, it can. Do it.
542  LogpWriteLogBufferToFile(&info, false);
543  status = LogpWriteMessageToFile(message, info);
544  }
545 #pragma warning(pop)
546  } else {
547  // No, it cannot. Set the prited bit if needed, and then buffer it.
548  if (do_DbgPrint) {
549  LogpSetPrintedBit(message, true);
550  }
551  status = LogpBufferMessage(message, &info);
552  LogpSetPrintedBit(message, false);
553  }
554  }
555 
556  // Can it safely be printed?
557  if (do_DbgPrint) {
558  const auto location_of_cr = strlen(message) - 2;
559  message[location_of_cr] = '\n';
560  message[location_of_cr + 1] = '\0';
561  DbgPrintEx(DPFLTR_DEFAULT_ID, DPFLTR_ERROR_LEVEL, "%s", message);
562  }
563  return status;
564 }

566 // Switch the current log buffer and save the contents of old buffer to the log
567 // file. This function does not flush the log file, so code should call
568 // LogpWriteMessageToFile() or ZwFlushBuffersFile() later.
569 _Use_decl_annotations_ static NTSTATUS LogpWriteLogBufferToFile(
570  LogBufferInfo *info, bool do_print) {
571  NT_ASSERT(info);
572  NT_ASSERT(KeGetCurrentIrql() == PASSIVE_LEVEL);
573 
574  auto status = STATUS_SUCCESS;
575 
576  // Enter a critical section and acquire a reader lock for info in order to
577  // write a log file safely.
578  ExEnterCriticalRegionAndAcquireResourceExclusive(&info->resource);
579 
580  // Acquire a spin lock for info.log_buffer(s) in order to switch its head
581  // safely.
582  KLOCK_QUEUE_HANDLE lock_handle = {};
583  KeAcquireInStackQueuedSpinLock(&info->spin_lock, &lock_handle);
584  const auto old_log_buffer = const_cast<char *>(info->log_buffer_head);
585  if (old_log_buffer[0]) {
586  info->log_buffer_head = (old_log_buffer == info->log_buffer1)
587  ? info->log_buffer2
588  : info->log_buffer1;
589  info->log_buffer_head[0] = '\0';
590  info->log_buffer_tail = info->log_buffer_head;
591  }
592  KeReleaseInStackQueuedSpinLock(&lock_handle);
593 
594  // Write all log entries in old log buffer.
595  IO_STATUS_BLOCK io_status = {};
596  for (auto current_log_entry = old_log_buffer; current_log_entry[0]; ) {
597  // Check the printed bit and clear it
598  const auto printed_out = LogpIsPrinted(current_log_entry);
599  LogpSetPrintedBit(current_log_entry, false);
600 
601  const auto current_log_entry_length = strlen(current_log_entry);
602  status = ZwWriteFile(info->log_file_handle, nullptr, nullptr, nullptr,
603  &io_status, current_log_entry,
604  static_cast<ULONG>(current_log_entry_length), nullptr,
605  nullptr);
606  if (!NT_SUCCESS(status)) {
607  // It could happen when you did not register IRP_SHUTDOWN and call
608  // LogIrpShutdownHandler() and the system tried to log to a file after
609  // a file system was unmounted.
610  NT_ASSERT(false);
611  }
612 
613  // Print it out if requested and the message is not already printed out
614  if (do_print && !printed_out) {
615  const auto location_of_cr = current_log_entry_length - 2;
616  current_log_entry[location_of_cr] = '\n';
617  current_log_entry[location_of_cr + 1] = '\0';
618  DbgPrintEx(DPFLTR_DEFAULT_ID, DPFLTR_ERROR_LEVEL, "%s",
619  current_log_entry);
620  }
621 
622  current_log_entry += current_log_entry_length + 1;
623  }
624  old_log_buffer[0] = '\0';
625 
626  ExReleaseResourceAndLeaveCriticalRegion(&info->resource);
627  return status;
628 }

650 // Buffer the log entry to the log buffer.
651 _Use_decl_annotations_ static NTSTATUS LogpBufferMessage(const char *message,
652  LogBufferInfo *info) {
653  NT_ASSERT(info);
654 
655  // Acquire a spin lock to add the log safely.
656  KLOCK_QUEUE_HANDLE lock_handle = {};
657  const auto old_irql = KeGetCurrentIrql();
658  if (old_irql < DISPATCH_LEVEL) {
659  KeAcquireInStackQueuedSpinLock(&info->spin_lock, &lock_handle);
660  } else {
661  KeAcquireInStackQueuedSpinLockAtDpcLevel(&info->spin_lock, &lock_handle);
662  }
663  NT_ASSERT(KeGetCurrentIrql() >= DISPATCH_LEVEL);
664 
665  // Copy the current log to the buffer.
666  SIZE_T used_buffer_size = info->log_buffer_tail - info->log_buffer_head;
667  auto status =
668  RtlStringCchCopyA(const_cast<char *>(info->log_buffer_tail),
669  kLogpBufferUsableSize - used_buffer_size, message);
670 
671  // Update info.log_max_usage if necessary.
672  if (NT_SUCCESS(status)) {
673  const auto message_length = strlen(message) + 1;
674  info->log_buffer_tail += message_length;
675  used_buffer_size += message_length;
676  if (used_buffer_size > info->log_max_usage) {
677  info->log_max_usage = used_buffer_size; // Update
678  }
679  } else {
680  info->log_max_usage = kLogpBufferSize; // Indicates overflow
681  }
682  *info->log_buffer_tail = '\0';
683 
684  if (old_irql < DISPATCH_LEVEL) {
685  KeReleaseInStackQueuedSpinLock(&lock_handle);
686  } else {
687  KeReleaseInStackQueuedSpinLockFromDpcLevel(&lock_handle);
688  }
689  return status;
690 }

726 // A thread runs as long as info.buffer_flush_thread_should_be_alive is true and
727 // flushes a log buffer to a log file every kLogpLogFlushIntervalMsec msec.
728 _Use_decl_annotations_ static VOID LogpBufferFlushThreadRoutine(
729  void *start_context) {
730  PAGED_CODE();
731  auto status = STATUS_SUCCESS;
732  auto info = reinterpret_cast<LogBufferInfo *>(start_context);
733  info->buffer_flush_thread_started = true;
734  HYPERPLATFORM_LOG_DEBUG("Log thread started (TID= %p).",
735  PsGetCurrentThreadId());
736 
737  while (info->buffer_flush_thread_should_be_alive) {
738  NT_ASSERT(LogpIsLogFileActivated(*info));
739  if (info->log_buffer_head[0]) {
740  NT_ASSERT(KeGetCurrentIrql() == PASSIVE_LEVEL);
741  NT_ASSERT(!KeAreAllApcsDisabled());
742  status = LogpWriteLogBufferToFile(info, true);
743  // Do not flush the file for overall performance. Even a case of
744  // bug check, we should be able to recover logs by looking at both
745  // log buffers.
746  }
747  LogpSleep(kLogpLogFlushIntervalMsec);
748  }
749  PsTerminateSystemThread(status);
750 }
