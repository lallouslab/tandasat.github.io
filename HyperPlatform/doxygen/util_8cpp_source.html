192 // Builds the physical memory ranges
193 _Use_decl_annotations_ static PhysicalMemoryDescriptor *
194 UtilpBuildPhysicalMemoryRanges() {
195  PAGED_CODE();
196 
197  const auto pm_ranges = MmGetPhysicalMemoryRanges();
198  if (!pm_ranges) {
199  return nullptr;
200  }
201 
202  PFN_COUNT number_of_runs = 0;
203  PFN_NUMBER number_of_pages = 0;
204  for (; ; ++number_of_runs) {
205  const auto range = &pm_ranges[number_of_runs];
206  if (!range->BaseAddress.QuadPart && !range->NumberOfBytes.QuadPart) {
207  break;
208  }
209  number_of_pages +=
210  static_cast<PFN_NUMBER>(BYTES_TO_PAGES(range->NumberOfBytes.QuadPart));
211  }
212  if (number_of_runs == 0) {
213  ExFreePoolWithTag(pm_ranges, 'hPmM');
214  return nullptr;
215  }
216 
217  const auto memory_block_size =
218  sizeof(PhysicalMemoryDescriptor) +
219  sizeof(PhysicalMemoryRun) * (number_of_runs - 1);
220  const auto pm_block =
221  reinterpret_cast<PhysicalMemoryDescriptor *>(ExAllocatePoolWithTag(
222  NonPagedPoolNx, memory_block_size, kHyperPlatformCommonPoolTag));
223  if (!pm_block) {
224  ExFreePoolWithTag(pm_ranges, 'hPmM');
225  return nullptr;
226  }
227  RtlZeroMemory(pm_block, memory_block_size);
228 
229  pm_block->number_of_runs = number_of_runs;
230  pm_block->number_of_pages = number_of_pages;
231 
232  for (auto run_index = 0ul; run_index < number_of_runs; run_index++) {
233  auto current_run = &pm_block->run[run_index];
234  auto current_block = &pm_ranges[run_index];
235  current_run->base_page = static_cast<ULONG_PTR>(
236  UtilPfnFromPa(current_block->BaseAddress.QuadPart));
237  current_run->page_count = static_cast<ULONG_PTR>(
238  BYTES_TO_PAGES(current_block->NumberOfBytes.QuadPart));
239  }
240 
241  ExFreePoolWithTag(pm_ranges, 'hPmM');
242  return pm_block;
243 }

251 // Execute a given callback routine on all processors in DPC_LEVEL. Returns
252 // STATUS_SUCCESS when all callback returned STATUS_SUCCESS as well. When
253 // one of callbacks returns anything but STATUS_SUCCESS, this function stops
254 // to call remaining callbacks and returns the value.
255 _Use_decl_annotations_ NTSTATUS
256 UtilForEachProcessor(NTSTATUS (*callback_routine)(void *), void *context) {
257  const auto number_of_processors = KeQueryActiveProcessorCount(nullptr);
258  for (ULONG processor_number = 0; processor_number < number_of_processors;
259  processor_number++) {
260  // Switch the current processor
261  const auto old_affinity = KeSetSystemAffinityThreadEx(
262  static_cast<KAFFINITY>(1ull << processor_number));
263  const auto old_irql = KeRaiseIrqlToDpcLevel();
264 
265  // Execute callback
266  const auto status = callback_routine(context);
267 
268  KeLowerIrql(old_irql);
269  KeRevertToUserAffinityThreadEx(old_affinity);
270  if (!NT_SUCCESS(status)) {
271  return status;
272  }
273  }
274  return STATUS_SUCCESS;
275 }

317 // Return true if the given address is accessible. It does not prevent a race

319 _Use_decl_annotations_ bool UtilIsAccessibleAddress(void *address) {
320 #if defined(_AMD64_)
321  const auto pxe = UtilpAddressToPxe(address);
322  const auto ppe = UtilpAddressToPpe(address);
323  if (!pxe->valid || !ppe->valid) {
324  return false;
325  }
326 #endif
327 
328  const auto is_x86_pae = UtilIsX86Pae();
329  const auto pde =
330  (is_x86_pae) ? UtilpAddressToPdePAE(address) : UtilpAddressToPde(address);
331  const auto pte =
332  (is_x86_pae) ? UtilpAddressToPtePAE(address) : UtilpAddressToPte(address);
333  if (!pde->valid) {
334  return false;
335  }
336  if (pde->large_page) {
337  return true; // A large page is always memory resident
338  }
339  if (!pte || !pte->valid) {
340  return false;
341  }
342  return true;
343 }

345 // Virtual Address Interpretation For Handling PTEs
346 //
347 // -- On x64
348 // Sign extension 16 bits
349 // Page map level 4 selector 9 bits
350 // Page directory pointer selector 9 bits
351 // Page directory selector 9 bits
352 // Page table selector 9 bits
353 // Byte within page 12 bits
354 // 11111111 11111111 11111000 10000000 00000011 01010011 00001010 00011000
355 // ^^^^^^^^ ^^^^^^^^ ~~~~~~~~ ~^^^^^^^ ^^~~~~~~ ~~~^^^^^ ^^^^~~~~ ~~~~~~~~
356 // Sign extension PML4 PDPT PD PT Offset
357 //
358 // -- On x86(PAE)
359 // Page directory pointer selector 2 bits
360 // Page directory selector 9 bits
361 // Page table selector 9 bits
362 // Byte within page 12 bits
363 // 10 000011011 000001101 001001110101
364 // ^^ ~~~~~~~~~ ^^^^^^^^^ ~~~~~~~~~~~~
365 // PDPT PD PT Offset
366 //
367 // -- On x86 and ARM
368 // Page directory selector 10 bits
369 // Page table selector 10 bits
370 // Byte within page 12 bits
371 // 1000001101 1000001101 001001110101
372 // ~~~~~~~~~~ ^^^^^^^^^^ ~~~~~~~~~~~~
373 // PD PT Offset
374 //
375 //
376 // x64 x86(PAE) x86 ARM
377 // Page map level 4 selector 9 - - -
378 // Page directory pointer selector 9 2 - -
379 // Page directory selector 9 9 10 10
380 // Page table selector 9 9 10 10
381 // Byte within page 12 12 12 12
382 //
383 // 6666555555555544444444443333333333222222222211111111110000000000
384 // 3210987654321098765432109876543210987654321098765432109876543210
385 // ----------------------------------------------------------------
386 // aaaaaaaaaaaaaaaabbbbbbbbbcccccccccdddddddddeeeeeeeeeffffffffffff x64
387 // ................................ccdddddddddeeeeeeeeeffffffffffff x86(PAE)
388 // ................................ddddddddddeeeeeeeeeeffffffffffff x86
389 // ................................ddddddddddeeeeeeeeeeffffffffffff ARM
390 //
391 // a = Sign extension, b = PML4, c = PDPT, d = PD, e = PT, f = Offset
392
