147 // Checks if the system supports virtualization
148 _Use_decl_annotations_ static bool VmpIsVmxAvailable() {
149  PAGED_CODE();
150 
151  // See: DISCOVERING SUPPORT FOR VMX
152  // If CPUID.1:ECX.VMX[bit 5]=1, then VMX operation is supported.
153  int cpu_info[4] = {};
154  __cpuid(cpu_info, 1);
155  const CpuFeaturesEcx cpu_features = {static_cast<ULONG_PTR>(cpu_info[2])};
156  if (!cpu_features.fields.vmx) {
157  HYPERPLATFORM_LOG_ERROR("VMX features are not supported.");
158  return false;
159  }
160 
161  // See: BASIC VMX INFORMATION
162  // The first processors to support VMX operation use the write-back type.
163  const Ia32VmxBasicMsr vmx_basic_msr = {UtilReadMsr64(Msr::kIa32VmxBasic)};
164  if (static_cast<memory_type>(vmx_basic_msr.fields.memory_type) !=
165  memory_type::kWriteBack) {
166  HYPERPLATFORM_LOG_ERROR("Write-back cache type is not supported.");
167  return false;
168  }
169 
170  // See: ENABLING AND ENTERING VMX OPERATION
171  Ia32FeatureControlMsr vmx_feature_control = {
172  UtilReadMsr64(Msr::kIa32FeatureControl)};
173  if (!vmx_feature_control.fields.lock) {
174  HYPERPLATFORM_LOG_INFO("The lock bit is clear. Attempting to set 1.");
175  const auto status = UtilForEachProcessor(VmpSetLockBitCallback, nullptr);
176  if (!NT_SUCCESS(status)) {
177  return false;
178  }
179  }
180  if (!vmx_feature_control.fields.enable_vmxon) {
181  HYPERPLATFORM_LOG_ERROR("VMX features are not enabled.");
182  return false;
183  }
184 
185  if (!EptIsEptAvailable()) {
186  HYPERPLATFORM_LOG_ERROR("EPT features are not fully supported.");
187  return false;
188  }
189  return true;
190 }

212 _Use_decl_annotations_ static SharedProcessorData *VmpInitializeSharedData() {
213  PAGED_CODE();
214 
215  const auto shared_data = reinterpret_cast<SharedProcessorData *>(
216  ExAllocatePoolWithTag(NonPagedPoolNx, sizeof(SharedProcessorData),
217  kHyperPlatformCommonPoolTag));
218  if (!shared_data) {
219  return nullptr;
220  }
221  RtlZeroMemory(shared_data, sizeof(SharedProcessorData));
222  HYPERPLATFORM_LOG_DEBUG("SharedData= %p", shared_data);
223 
224  // Set up the MSR bitmap
225  const auto msr_bitmap = ExAllocatePoolWithTag(NonPagedPoolNx, PAGE_SIZE,
226  kHyperPlatformCommonPoolTag);
227  if (!msr_bitmap) {
228  ExFreePoolWithTag(shared_data, kHyperPlatformCommonPoolTag);
229  return nullptr;
230  }
231  RtlZeroMemory(msr_bitmap, PAGE_SIZE);
232  shared_data->msr_bitmap = msr_bitmap;
233 
234  // Activate VM-exit for RDMSR against all MSRs
235  const auto bitmap_read_low = reinterpret_cast<UCHAR *>(msr_bitmap);
236  const auto bitmap_read_high = bitmap_read_low + 1024;
237  RtlFillMemory(bitmap_read_low, 1024, 0xff); // read 0 - 1fff
238  RtlFillMemory(bitmap_read_high, 1024, 0xff); // read c0000000 - c0001fff
239 
240  // But ignore IA32_MPERF (000000e7) and IA32_APERF (000000e8)
241  RTL_BITMAP bitmap_read_low_header = {};
242  RtlInitializeBitMap(&bitmap_read_low_header,
243  reinterpret_cast<PULONG>(bitmap_read_low), 1024 * 8);
244  RtlClearBits(&bitmap_read_low_header, 0xe7, 2);
245 
246  // But ignore IA32_GS_BASE (c0000101) and IA32_KERNEL_GS_BASE (c0000102)
247  RTL_BITMAP bitmap_read_high_header = {};
248  RtlInitializeBitMap(&bitmap_read_high_header,
249  reinterpret_cast<PULONG>(bitmap_read_high), 1024 * 8);
250  RtlClearBits(&bitmap_read_high_header, 0x101, 2);
251 
252  // Set up EPT
253  shared_data->ept_data = EptInitialization();
254  if (!shared_data->ept_data) {
255  ExFreePoolWithTag(msr_bitmap, kHyperPlatformCommonPoolTag);
256  ExFreePoolWithTag(shared_data, kHyperPlatformCommonPoolTag);
257  return nullptr;
258  }
259  return shared_data;
260 }

277 _Use_decl_annotations_ static void VmpInitializeVm(
278  ULONG_PTR guest_stack_pointer, ULONG_PTR guest_instruction_pointer,
279  void *context) {
280  const auto shared_data = reinterpret_cast<SharedProcessorData *>(context);
281  if (!shared_data) {
282  return;
283  }
284 
285  // Allocate related structures
286  const auto processor_data =
287  reinterpret_cast<ProcessorData *>(ExAllocatePoolWithTag(
288  NonPagedPoolNx, sizeof(ProcessorData), kHyperPlatformCommonPoolTag));
289  if (!processor_data) {
290  return;
291  }
292  RtlZeroMemory(processor_data, sizeof(ProcessorData));
293 
294  const auto vmm_stack_limit = UtilAllocateContiguousMemory(KERNEL_STACK_SIZE);
295  const auto vmcs_region =
296  reinterpret_cast<VmControlStructure *>(ExAllocatePoolWithTag(
297  NonPagedPoolNx, kVmxMaxVmcsSize, kHyperPlatformCommonPoolTag));
298  const auto vmxon_region =
299  reinterpret_cast<VmControlStructure *>(ExAllocatePoolWithTag(
300  NonPagedPoolNx, kVmxMaxVmcsSize, kHyperPlatformCommonPoolTag));
301 
302  // Initialize the management structure
303  processor_data->vmm_stack_limit = vmm_stack_limit;
304  processor_data->vmcs_region = vmcs_region;
305  processor_data->vmxon_region = vmxon_region;
306 
307  if (!vmm_stack_limit || !vmcs_region || !vmxon_region) {
308  goto ReturnFalse;
309  }
310  RtlZeroMemory(vmm_stack_limit, KERNEL_STACK_SIZE);
311  RtlZeroMemory(vmcs_region, kVmxMaxVmcsSize);
312  RtlZeroMemory(vmxon_region, kVmxMaxVmcsSize);
313 
314  // Initialize stack memory for VMM like this:
315  //
316  // (High)
317  // +------------------+ <- vmm_stack_region_base (eg, AED37000)
318  // | processor_data |
319  // +------------------+ <- vmm_stack_data (eg, AED36FFC)
320  // | MAXULONG_PTR |
321  // +------------------+ <- vmm_stack_base (initial SP)(eg, AED36FF8)
322  // | | v
323  // | (VMM Stack) | v (grow)
324  // | | v
325  // +------------------+ <- vmm_stack_limit (eg, AED34000)
326  // (Low)
327  const auto vmm_stack_region_base =
328  reinterpret_cast<ULONG_PTR>(vmm_stack_limit) + KERNEL_STACK_SIZE;
329  const auto vmm_stack_data = vmm_stack_region_base - sizeof(void *);
330  const auto vmm_stack_base = vmm_stack_data - sizeof(void *);
331  HYPERPLATFORM_LOG_DEBUG("VmmStackTop= %p", vmm_stack_limit);
332  HYPERPLATFORM_LOG_DEBUG("VmmStackBottom= %p", vmm_stack_region_base);
333  HYPERPLATFORM_LOG_DEBUG("VmmStackData= %p", vmm_stack_data);
334  HYPERPLATFORM_LOG_DEBUG("ProcessorData= %p stored at %p", processor_data,
335  vmm_stack_data);
336  HYPERPLATFORM_LOG_DEBUG("VmmStackBase= %p", vmm_stack_base);
337  HYPERPLATFORM_LOG_DEBUG("GuestStackPointer= %p", guest_stack_pointer);
338  HYPERPLATFORM_LOG_DEBUG("GuestInstPointer= %p", guest_instruction_pointer);
339  *reinterpret_cast<ULONG_PTR *>(vmm_stack_base) = MAXULONG_PTR;
340  *reinterpret_cast<ProcessorData **>(vmm_stack_data) = processor_data;
341 
342  processor_data->shared_data = shared_data;
343  InterlockedIncrement(&processor_data->shared_data->reference_count);
344 
345  // Set up VMCS
346  if (!VmpEnterVmxMode(processor_data)) {
347  goto ReturnFalse;
348  }
349  if (!VmpInitializeVMCS(processor_data)) {
350  goto ReturnFalseWithVmxOff;
351  }
352  if (!VmpSetupVMCS(processor_data, guest_stack_pointer,
353  guest_instruction_pointer, vmm_stack_base)) {
354  goto ReturnFalseWithVmxOff;
355  }
356 
357  // Do virtualize the processor
358  VmpLaunchVM();
359 
360 // Here is not be executed with successful vmlaunch. Instead, the context
361 // jumps to an address specified by guest_instruction_pointer.
362 
363 ReturnFalseWithVmxOff:;
364  __vmx_off();
365 
366 ReturnFalse:;
367  VmpFreeProcessorData(processor_data);
368 }

371 _Use_decl_annotations_ static bool VmpEnterVmxMode(
372  ProcessorData *processor_data) {
373  // Apply FIXED bits
374  const Cr0 cr0_fixed0 = {UtilReadMsr(Msr::kIa32VmxCr0Fixed0)};
375  const Cr0 cr0_fixed1 = {UtilReadMsr(Msr::kIa32VmxCr0Fixed1)};
376  Cr0 cr0 = {__readcr0()};
377  cr0.all &= cr0_fixed1.all;
378  cr0.all |= cr0_fixed0.all;
379  __writecr0(cr0.all);
380 
381  const Cr4 cr4_fixed0 = {UtilReadMsr(Msr::kIa32VmxCr4Fixed0)};
382  const Cr4 cr4_fixed1 = {UtilReadMsr(Msr::kIa32VmxCr4Fixed1)};
383  Cr4 cr4 = {__readcr4()};
384  cr4.all &= cr4_fixed1.all;
385  cr4.all |= cr4_fixed0.all;
386  __writecr4(cr4.all);
387 
388  // Write a VMCS revision identifier
389  const Ia32VmxBasicMsr vmx_basic_msr = {UtilReadMsr64(Msr::kIa32VmxBasic)};
390  processor_data->vmxon_region->revision_identifier =
391  vmx_basic_msr.fields.revision_identifier;
392 
393  auto vmxon_region_pa = UtilPaFromVa(processor_data->vmxon_region);
394  if (__vmx_on(&vmxon_region_pa)) {
395  return false;
396  }
397 
398  UtilInveptAll();
399  return true;
400 }
