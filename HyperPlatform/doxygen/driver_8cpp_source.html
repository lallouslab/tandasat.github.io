137 // Test if the system is one of supported OS versions
138 _Use_decl_annotations_ bool DriverpIsSuppoetedOS() {
139  PAGED_CODE();
140 
141  RTL_OSVERSIONINFOW os_version = {};
142  auto status = RtlGetVersion(&os_version);
143  if (!NT_SUCCESS(status)) {
144  return false;
145  }
146  if (os_version.dwMajorVersion != 6 && os_version.dwMajorVersion != 10) {
147  return false;
148  }
149  // 4-gigabyte tuning (4GT) should not be enabled

152  return false;
